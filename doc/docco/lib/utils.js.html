        const remappedHandlers = _.mapValues( handlers, handler => {

                handler = {
                    value: handler,
                };
            }
            let defaults = {
                enumerable: true,
            };
            if ( !handler.hasOwnProperty( 'get' )) {
                defaults.writable = false;
            }
            _.defaults( handler, defaults );
            return handler;
        });
        return Object.defineProperties( subject, remappedHandlers );
