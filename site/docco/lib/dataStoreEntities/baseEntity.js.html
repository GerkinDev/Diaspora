class DataStoreEntity {
    constructor( entity, dataSource ) {
        if ( !c.assigned( entity )) {
            return undefined;
        }
        Object.defineProperties( this, {
            dataSource: {
                value:        dataSource,
                enumerable:   false,
                configurable: false,
            },
        });
        _.assign( this, entity );
    }
    toObject() {
        return _.omit( this, 'dataSource' );
    }
}

module.exports = DataStoreEntity;
